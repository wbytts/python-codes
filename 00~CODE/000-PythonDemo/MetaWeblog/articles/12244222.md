#include <opencv2/opencv.hpp>

int main(int argc, char** argv)

        return -1;

    namedWindow("input", WINDOW_AUTOSIZE);
    imshow("input", src);

    int h = src.rows;
    int ch = src.channels();

    printf("w: %d, h: %d, ch: %d\n", w, h, ch);

    vector<int> hist(256);
    for (int i = 0; i < 256; i++) {
        hist[i] = 0;
    }
